Value at Risk Model


import numpy as np
import pandas as pd
from scipy import stats

# Historical simulation VaR
def calculate_var_historical(returns, confidence=0.05):
    """Calculate Value at Risk using historical simulation"""
    return np.percentile(returns, confidence * 100)

# Monte Carlo VaR
def calculate_var_monte_carlo(portfolio_value, mu, sigma, days=252, simulations=10000):
    """Monte Carlo simulation for VaR"""
    dt = 1/days
    random_shocks = np.random.normal(0, 1, simulations)
    returns = mu * dt + sigma * np.sqrt(dt) * random_shocks
    return portfolio_value * np.percentile(returns, 5)

# Example usage
portfolio = 100_000_000  # $100M portfolio
daily_returns = np.random.normal(0.001, 0.02, 1000)

var_95 = calculate_var_historical(daily_returns)
print(f"95% VaR: ${var_95 * portfolio:,.2f}")

Daily VaR: $3.2M (95% confidence)

Technical explanation for quant stakeholders:

  • Two complementary VaR methodologies implemented
  • Historical simulation uses actual price movements
